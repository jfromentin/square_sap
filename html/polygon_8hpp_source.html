90inline void Polygon::set_length(size_t l) {
91 length = l;
92 for(size_t i = 0; i < l / 2 - 1; ++ i) {
93 steps[i] = Right;
94 steps[i + l / 2] = Left;
95 }
96 steps[l / 2 - 1] = Up;
97 steps[l - 1] = Down;
