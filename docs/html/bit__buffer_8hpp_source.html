116template<class T> inline void BitBuffer<Out, T>::write_bit(bool b) {
117
118 // We add the value of head to current_byte if b is true
119 if(b) current_byte |= head;
120 // We multiply the head by 2, to design the next bit positoin
121 head <<= 1;
122 // If head = 0 = 2^8, we exceed the capacity of a byte and we call add_byte
123 if (head == 0) write_byte();
124}

126template<class T> inline void BitBuffer<Out, T>::write_byte() {
127 // Write Byte v in the next available place in buffer
128 buffer[size ++] = current_byte;
129 // Initialise v and head
130 current_byte = 0;
131 head = 1;
132 // If the capacity of buffer is reached, we write it to output
133 if (size == capacity) {
134 // Flush current buffer to output object
135 output->write(buffer, size);
136 size = 0;
137 }
138}

157template<class T> inline void BitBuffer<Out, T>::close() {
158 // If head is not equal to 1, we must add bit '0' to complete the current byte
159 while (head != 1) {
160 write_bit(false);
161 }
162 if (size != 0) output->write(buffer, size);
163 // Close the output object
164 output->close();
165}

179template<class T> inline bool BitBuffer<In, T>::read_bit() {
180 // If we can read nothing more in current_byte, head is 0 = 2^8
181 if (head == 0) {
182 // We want rad the next byte in the internal buffer
183 ++ pos;
184 // If it is not possible
185 if (pos >= size) {
186 // Fill the internal buffer with at most capacity bytes from the input
187 // size if then set to number of bytes actually read and pos is set to 0
188 read_bytes();
189 // Check we have read something
190 assert(size != 0);
191 }
192 // Set current_byte to the new pointed byte in the internal buffer
193 current_byte = buffer[pos];
194 // Next bit to read is at position 0 and we set haed to 2^0 = 1
195 head = 1;
196 }
197 // Read the value of the bit given by head
198 bool b = (head & current_byte);
199 // Shift the head (multiply it by 2)
200 head <<= 1;
201 // Return the read bit
202 return b;
203}

205template<class T> inline void BitBuffer<In, T>::read_bytes() {
206 // Try to fill the internal buffer with capacity bytes read from the input
207 // the actual number of bytes read is stored in size
208 size = input->read(buffer, capacity);
209 // We point to the first byte of the internal buffer
210 pos = 0;
211}

213template<class T> inline size_t BitBuffer<In, T>::read_int(size_t nb) {
214 size_t res = 0;
215 for (size_t i = 0; i < nb; ++ i) {
216 if (read_bit()) res += (1 << i);
217 }
218 return res;
219}
