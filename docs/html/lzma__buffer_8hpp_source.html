102template<class T> void LzmaBuffer<Out, T>::write(Byte* buffer, size_t size) {
103 // We assume that size is less than the input buffer capacity
104 assert(size <= capacity);
105 // Run LZMA algorithm until we reach the end of the input buffer
106 lzma_action action = LZMA_RUN;
107 // Set buffer as input buffer (inflate) of lzma algorithm
108 strm.next_in = buffer;
109 strm.avail_in = size;
110 // Loop until input buffer is empty
111 while (strm.avail_in) {
112 // Call lzma defalte algorithm
113 lzma_ret ret = lzma_code(&strm, action);
114 // Test if an error occured
115 if (ret != LZMA_OK) {
116 cerr << "[LZMA error] " << message(ret) << endl;
117 exit(0);
118 }
119 // Test if the ouput buffer is full
120 if (strm.avail_out == 0) {
121 // If it the case, flush the output buffer in output object
122 output -> write(output_buffer, capacity);
123 // Reset the output buffer
124 strm.next_out = output_buffer;
125 strm.avail_out = capacity;
126 }
127 }
128}

130template<class T> void LzmaBuffer<Out, T>::close() {
131 assert(strm.avail_in == 0);
132 // Write the remaining data
133 output -> write(output_buffer, capacity - strm.avail_out);
134 strm.next_out = output_buffer;
135 strm.avail_out = capacity;
136 lzma_ret ret;
137 do {
138 ret = lzma_code(&strm, LZMA_FINISH);
139 if (ret != LZMA_OK and ret != LZMA_STREAM_END) {
140 cerr << "[LZMA error] " << message(ret) << endl;
141 }
142 output -> write(output_buffer, capacity - strm.avail_out);
143 } while(ret != LZMA_STREAM_END);
144 output -> close();
145}

166template<class T> size_t LzmaBuffer<In, T>::read(Byte* buffer, size_t size) {
167 // Set output buffer
168 strm.next_out = buffer;
169 strm.avail_out = size;
170 // Loop until we can add data to output buffer
171 while(strm.avail_out > 0) {
172 // Test is there is available data in input buffer
173 if(strm.avail_in == 0) {
174 // If no, fill the input buffer
175 strm.next_in = input_buffer;
176 strm.avail_in = input->read(input_buffer, capacity);
177 // Test is there is now available data in
178 if(strm.avail_in == 0) {
179 // If no, tell LZMA algorithm that we want to finish
180 action = LZMA_FINISH;
181 }
182 }
183 // Call the LZMA algoritm
184 lzma_ret ret = lzma_code(&strm, action);
185 // Check if an error occured
186 if (ret == LZMA_STREAM_END ) {
187 // Enf of the input stream reached, quit the loop
188 break;
189 }
190 if (ret == LZMA_BUF_ERROR) {
191 break;
192 }
193
194 // Test if an error occured
195 if (ret != LZMA_OK ) {//and ret != LZMA_BUF_ERROR) {
196 cerr << "[LZMA error] " << message(ret) << endl;
197 }
198 }
199 // Compute the number of bytes actually added in output uffer
200 size_t s = size - strm.avail_out;
201 // Return this value
202 return s;
203}
