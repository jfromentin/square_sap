83template<class T> inline BitBuffer<Out, T>::BitBuffer(T& o):output(o){
84 // The internal buffer is empty

89 head = 1;

92template<class T> inline void BitBuffer<Out, T>::write_bit(bool b) {
93 // We add the value of head to current_byte if b is true
94 if(b) current_byte |= head;
95 // We multiply the head by 2, to design the next bit positoin
96 head <<= 1;
97 // If head = 0 = 2^8, we exceed the capacity of a byte and we call add_byte
98 if (head == 0) write_byte();
99}

101template<class T> inline void BitBuffer<Out, T>::write_byte() {
102 // Write Byte v in the next available place in buffer
103 buffer[size ++] = current_byte;
104 // Initialise v and head
105 current_byte = 0;
106 head = 1;
107 // If the capacity of buffer is reached, we write it to output
108 if (size == capacity) {
109
110 // Flush current buffer to output object
111 output.write(buffer, size);
112 size = 0;
113 }
114}

116template<class T> inline void BitBuffer<Out, T>::write_int(size_t v, size_t nb) {
117 // Write nb bits accordinglyto the value of v
118 for (size_t i = 0; i< nb; ++ i) {
119 write_bit(v & (1 << i));

123template<class T> inline void BitBuffer<Out, T>::close() {
124 // If head is not equal to 1, we must add bit '0' to complete the current byte
125 while (head != 1) {
126 write_bit(false);
127 }
128 // If s is equal to capcity, wryte_byte writes the buffer
129 // Otherwise, we do it now
130 if (size < capacity) {
131 output.write(buffer, size);
132 }
133
134 // Close the output object
135 output.close();
136}

146template<class T> inline bool BitBuffer<In, T>::read_bit() {
147 // If we can read nothing more in current_byte, head is 0 = 2^8
148 if (head == 0) {
149 // We want rad the next byte in the internal buffer
150 ++ pos;
151 // If it is not possible
152 if (pos >= size) {
153 // Fill the internal buffer with at most capacity bytes from the input
154 // size if then set to number of bytes actually read and pos is set to 0
155 read_bytes();
156 // Check we have read something
157 assert(size != 0);
158 }
159 // Set current_byte to the new pointed byte in the internal buffer
160 current_byte = buffer[pos];
161 // Next bit to read is at position 0 and we set haed to 2^0 = 1
162 head = 1;
163 }
164 // Read the value of the bit given by head
165 bool b = (head & current_byte);
166 // Shift the head (multiply it by 2)
167 head <<= 1;
168 // Return the read bit
169 return b;
170}

172template<class T> inline void BitBuffer<In, T>::read_bytes() {
173 // Try to fill the internal buffer with capacity bytes read from the input
174 // the actual number of bytes read is stored in size
175 size = input.read(buffer, capacity);
176 // We point to the first byte of the internal buffer
177 pos = 0;
178}

180template<class T> inline size_t BitBuffer<In, T>::read_int(size_t nb) {
181 size_t res = 0;
182 for (size_t i = 0; i < nb; ++ i) {
183 if (read_bit()) res += (1 << i);
184 }
185 return res;
186}
