74template<class T> LzmaBuffer<Out, T>::LzmaBuffer(T& o):output(o){
75 // Initialising the internal LZMA encoder structure using default preset and 64 bit checksum
76 strm = LZMA_STREAM_INIT;
77 lzma_ret ret = lzma_easy_encoder(&strm, LZMA_PRESET_DEFAULT, LZMA_CHECK_CRC64);
78 // Set output buffer for inflated data
79 strm.next_out = output_buffer;
80 strm.avail_out = capacity;
81 // Check if an error occured
82 if (ret != LZMA_OK) {
83 cerr << "[LZMA error] " << message(ret) << endl;
84 }

87template<class T> void LzmaBuffer<Out, T>::write(Byte* buffer, size_t size) {
88 // We assume that size is less than the input buffer capacity
89 assert(size <= capacity);
90 // Run LZMA algorithm until we reach the end of the input buffer
91 lzma_action action = LZMA_RUN;
92 // Set buffer as input buffer (inflate) of lzma algorithm
93 strm.next_in = buffer;
94 strm.avail_in = size;
95 // Loop until input buffer is empty
96 while (strm.avail_in) {
97 // Call lzma defalte algorithm
98 lzma_ret ret = lzma_code(&strm, action);
99 // Test if an error occured
100 if (ret != LZMA_OK) cerr << "[LZMA error] " << message(ret) << endl;
101 // Test if the ouput buffer is full
102 if (strm.avail_out == 0) {
103 // If it the case, flush the output buffer in output object
104 output.write(output_buffer, capacity);
105 // Reset the output buffer
106 strm.next_out = output_buffer;
107 strm.avail_out = capacity;
108 }
109 }
110}

112template<class T> void LzmaBuffer<Out, T>::close() {
113 // Write the remaining data
114 lzma_ret ret = lzma_code(&strm, LZMA_FINISH);
115 output.write(output_buffer, capacity - strm.avail_out);
116 output.close();
117}

131template<class T> size_t LzmaBuffer<In, T>::read(Byte* buffer, size_t size) {
132 // Set output buffer
133 strm.next_out = buffer;
134 strm.avail_out = size;
135
136 // Loop until we can add data to output buffer
137 while(strm.avail_out > 0) {
138
139 // Test is there is available data in input buffer
140 if(strm.avail_in == 0) {
141 // If no, fill the input buffer
142 strm.next_in = input_buffer;
143 strm.avail_in = input.read(input_buffer, capacity);
144 // Test is there is now available data in
145 if(strm.avail_in == 0) {
146 // If no, tell LZMA algorithm that we want to finish
147 action = LZMA_FINISH;
148 }
149 }
150 // Call the LZMA algoritm
151 lzma_ret ret = lzma_code(&strm, action);
152 // Check if an error occured
153 if (ret == LZMA_STREAM_END) {
154 // Enf of the input stream reached, quit the loop
155 break;
156 }
157 // Test if an error occured
158 if (ret != LZMA_OK) {
159 cerr << "[LZMA error] (" << ret << ") " << message(ret) << endl;
160 exit(0);
161 }
162 }
163 // Compute the number of bytes actually added in output uffer
164 size_t s = size - strm.avail_out;
165 // Return this value
166 return s;
167}
